CREATE TABLE SRLOG (
ID INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY, 
RFID VARCHAR(30) NOT NULL, 
USERNAME VARCHAR(20) NOT NULL, 
LDR VARCHAR(20) NOT NULL, 
PIR VARCHAR(20) NOT NULL,
SM VARCHAR(20) NOT NULL,
LAMP_STATUS VARCHAR(20) NOT NULL, 
LIGHT_STATUS VARCHAR(20) NOT NULL, 
ROOM_STATUS VARCHAR(20) NOT NULL,
DT DATETIME(6) NOT NULL
);
